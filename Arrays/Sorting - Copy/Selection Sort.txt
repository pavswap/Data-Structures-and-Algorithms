import java.util.*;

class Practice extends Input {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        Practice in = new Practice();

        System.out.print("Enter a Array : ");
        int[] ar = in.input(sc.nextLine());

        int[] sorted = selectionSort(ar, ar.length);
        System.out.println(Arrays.toString(sorted));
    }
    static int[] selectionSort(int[] ar, int n) {

        int maxindex = 0;
        for(int end = n - 1; end > 0; end--) {

            maxindex = findMax(ar, end);
            if(maxindex != end) {

                int swap = ar[end];
                ar[end] = ar[maxindex];
                ar[maxindex] = swap;
            }
        }
        return ar;
    }
    static int findMax(int[] ar, int end) {

        int max = ar[0], maxindex = 0;
        for(int i = 1; i <= end; i++) {
            if(ar[i] > max) {
                max = ar[i];
                maxindex = i;
            }
        }
        return maxindex;
    }

}
