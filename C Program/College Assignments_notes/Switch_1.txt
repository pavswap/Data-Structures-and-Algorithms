#include<stdio.h>
int main()
{
    printf("Enter 2 Numbers :\n");
    int a,b;
    scanf("%d %d", &a, &b);
    char ch;
    printf("Enter the operator\n");
    scanf(" %c ", &ch);
    switch(ch)
    {
        case '+': printf("Sum : %d \n", a+b);
        break;
        case '-': printf("Difference : %d \n", a-b);
        break;
        case '*': printf("Multiplication : %d \n", a*b);
        break;
        case '/': printf("Division : %d \n", a/b);
        break;
        default :printf("Wrong Input");
    }
}
