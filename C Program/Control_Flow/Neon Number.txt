#include <stdio.h>
#include <math.h>
int main()
{
    int n,a=0,s=0;
    printf("Enter a Number\n");
    scanf("%d", &n);
    int n_sq=pow(n,2);
    while(n_sq>0)
    {
        a=n_sq%10;
        n_sq=n_sq/10;
        s=s+a;
    }
    if(s==n)
    {
        printf("%d is a Neon Number\n", n);
    }
    else
    {
        printf("%d is not a Neon Number\n", n);
    }
}
