#include<stdio.h>
#include<math.h>
int main() 
{
    int s=0;
    int numeator = 10;
    int denominator = 9;
    float result = (float) numeator / denominator;
    printf("Enter the Number of Terms in the Series :");
    int n;
    scanf("%d", &n);
    for(int i=0;i<n;i++)
    {
        printf("%.0f ", (result*pow(10,i)));
        int k=result*pow(10,i);
        s=s+k;
    }
    printf("\n\nSum of the Series : %.0f", s);
}
