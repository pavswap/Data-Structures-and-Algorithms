#include<stdio.h>
#include <string.h>
int main()
{
    printf("Enter a Sentence\n");
    char sent[100];
    int k=0,j=0,m=0;
    fgets(sent, 100, stdin);
    int len = strlen(sent);
    if (sent[len - 1] == '\n') 
    {
        sent[len - 1] = '\0';
        len--; 
    }
    for(int i=0;i<len;i++)
    {
        char a=sent[i];
        if(a>=65 && a<=90 || a>=97 && a<=122)
        {
            k++;
        }
        else if(a>=48 && a<=57)
        {
            j++;
        }
        else if(a>=32 && a<=47 || a>=58 && a<=64 || a>=91 && a<=96)
        {
            m++;
        }
    }
    printf("\nNumber of Characters : %d", k);
    printf("\nNumber of Digits : %d", j);
    printf("\nNumber of Special Characters : %d", m);
}
