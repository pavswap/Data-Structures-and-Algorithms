#include<stdio.h>
#include<string.h>
#include<ctype.h>
int main()
{
    char s[100];
    scanf("%s", s);int vol=0,cons=0,dig=0,spc=0;
    for(int i=0;i<strlen(s);i++)
    {
        if(isalpha(s[i]))
        {
            if(s[i]=='a' || s[i]=='e' || s[i]=='i' || s[i]=='o' || s[i]=='u')
            {
                vol++;
            }
            else
            {
                cons++;
            }
        }
        else if(isdigit(s[i]))
        {
            dig++;
        }
        else
        {
            spc++;
        }
        
    }
    printf("Vowels: %d\nConsonants: %d\nDigits: %d\nSpecial Characters: %d",vol, cons, dig, spc);
}
