#include<stdio.h>
int main()
{
    printf("Enter the length of the String\n");
    int n;
    scanf("%d", &n);
    char string[n];
    printf("Enter the String\n");
    for(int i=0;i<n;i++)
    {
        scanf("%c", &string[i]);
    }
    printf("\n\n");
    for(int i=0;i<n;i++)
    {
        printf("%c", string[i]);
    }
}
