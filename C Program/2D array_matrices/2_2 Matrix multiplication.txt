#include<stdio.h>
int main()
{
   printf("Enter Values in the first Matrice\n");
   int mat1[2][2];
   for(int i=0;i<2;i++)
   {
    for(int j=0;j<2;j++)
    {
        printf("Enter Value for The First Matrice at positon (%d,%d)\n", i,j);
        scanf("%d", &mat1[i][j]);
    }
   }
   printf("Enter Values in the Second Matrice\n");
   int mat2[2][2];
   for(int i=0;i<2;i++)
   {
    for(int j=0;j<2;j++)
    {
        printf("Enter Value for The Second Matrice at positon (%d,%d)\n", i,j);
        scanf("%d", &mat2[i][j]);
    }
   }
   int mul_mat[2][2];
   mul_mat[0][0]=mat1[0][0]*mat2[0][0]+mat1[0][1]*mat2[1][0];
   mul_mat[0][1]=mat1[1][0]*mat2[0][0]+mat1[1][0]*mat2[1][1];
   mul_mat[1][0]=mat1[0][0]*mat2[0][1]+mat1[0][1]*mat2[1][1];
   mul_mat[1][1]=mat1[1][0]*mat2[0][1]+mat1[1][1]*mat2[1][1];
   printf("\nMultiplied Matrix\n");
   for(int i=0;i<2;i++)
   {
    printf("\n\n");
    for(int j=0;j<2;j++)
    {
        printf("%d  ", mul_mat[i][j]);
    }
   }
}
