#include <stdio.h>
 int main()
 {
    printf("Enter the Numbers of Rows for the  Matrice\n");
    int r;
    scanf("%d", &r);
    printf("Enter the Numbers of Colums for the  Matrice\n");
    int c;
    scanf("%d", &c);
    int mat[r][c];
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            printf("Enter Value int the first matrice for position [%d][%d] :", i, j);
            scanf("%d", &mat[i][j]);
        }
    }
    int copy[r][c];
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
           copy[i][j]=mat[j][i];
        };
    }
    printf("Transposed Array :\n");
    for(int i=0;i<r;i++)
    {
        printf("\n");
        for(int j=0;j<c;j++)
        {
            printf("%d ", copy[i][j]);
        }
    }
 }
