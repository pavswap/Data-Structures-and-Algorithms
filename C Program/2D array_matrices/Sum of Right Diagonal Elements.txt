#include <stdio.h>
 int main()
 {
    printf("Enter the Numbers of Rows for the  Matrice\n");
    int r;
    scanf("%d", &r);
    printf("Enter the Numbers of Colums for the  Matrice\n");
    int c;
    scanf("%d", &c);
    int mat[r][c];
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            printf("Enter Value int the first matrice for position [%d][%d] :", i, j);
            scanf("%d", &mat[i][j]);
        }
    }
    int s=0;
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
           if(i==j)
           {
            s=s+mat[i][j];
           }
        }
    }
    printf("Sum of Right Diagonal Elements : %d", s);
 }
