#include<stdio.h>
int main()
{
    printf("Enter the size of the array\n");
    int n;
    scanf("%d", &n);
    int arr[n];
    for(int i=0;i<n;i++)
    {
        printf("Enter value in the array\n");
        scanf("%d", &arr[i]);
    }
    for(int i=n;i>0;i--)
    {
        printf("%d\n", arr[i]);
    }
    
    
}
