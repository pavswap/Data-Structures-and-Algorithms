#include<stdio.h>
int main()
{
    printf("Enter the size of the first array \n");
    int n;
    scanf("%d", &n);
    int ar[n];
    for(int i=0;i<n;i++)
    {
        printf("Enter Value for Position %d\n", i);
        scanf("%d", &ar[i]);
    }
    printf("Enter the size of the second array\n");
    int n2;
    scanf("%d", &n2);
    int ar2[n2];
    for(int i=0;i<n2;i++)
    {
        printf("Enter Value for Position %d\n", i);
        scanf("%d", &ar2[i]);
    }
    printf("\n[ ");
    for(int i=0;i<n;i++)
    {
        printf("%d ", ar[i]);
    }
    printf("] + [ ");
    for(int i=0;i<n2;i++)
    {
        printf("%d ", ar2[i]);
    }
    printf("]\n");
    int n3=n+n2;
    int ar3[n3];
    for(int i=0;i<n;i++)
    {
        ar3[i]=ar[i];
    }
    for(int i=n;i<n3;i++)
    {
        ar3[i]=ar2[i-n];
    }
    printf("\nMerged Array\n[ ");
    for(int i=0;i<n3;i++)
    {
        printf("%d ", ar3[i]);
    }
    printf("]\n");
    for(int i=0;i<n3-1;i++)
    {
        for(int j=0;j<n3-i-1;j++)
        {
            if(ar3[j]>ar3[j+1])
            {
                int swap=ar3[j];
                ar3[j]=ar3[j+1];
                ar3[j+1]=swap;
            }
        }
    }
    printf("\nSorted and Merged Array\n[ ");
    for(int i=0;i<n3;i++)
    {
        printf("%d ", ar3[i]);
    }
    printf("]");
}
