#include<stdio.h>
int main()
{
    int n;
    printf("Enter a Number\n");
    scanf("%d", &n);
    int ar[n];
    for(int i=0;i<n;i++)
    {
        printf("Enter the Values in the Array\n");
        scanf("%d", &ar[i]);
    }
    printf("Enter 1 to Add Element\nEnter 2 to Remove an Element\n");
    int k;
    scanf("%d", &k);
    if(k==1)
    {
        printf("Enter the element you want to add\n");
        int el1;
        scanf("%d", &el1);
        int ar1[n+1];
        for(int i=0;i<n;i++)
        {
            ar1[i]=ar[i];
        }
        ar1[n]=el1;
        printf("Here is the New Array\n");
        for(int i=0;i<n+1;i++)
        {
            printf("%d\n", ar1[i]);
        }
    }
    else if(k==2)
    {
        printf("Enter the index number you want to remove\n");
        int indx;
        scanf("%d", &indx);
        ar[indx]=0;
        int swap=ar[indx];
        ar[indx]=ar[n-1];
        ar[n-1]=swap;
        printf("Here is the new Array\n");
        for(int i=0;i<n-1;i++)
    {
        printf("%d\n", ar[i]);
    }
    }
    else
    {
        printf("Wrong Input\n Run Code again");
    }
}
