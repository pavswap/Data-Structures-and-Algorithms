#include<stdio.h>
int main()
{
    int n;
    printf("Enter the Size of the Array\n");
    scanf("%d", &n);
    int n2=n;
    int ar[n];
    for(int i=0;i<n;i++)
    {
        printf("Enter for position %d\n", i);
        scanf("%d", &ar[i]);
    }
    printf("\nEntered Array : ");
    for(int i=0;i<n;i++)
    {
        printf("%d ", ar[i]);
    }
    int i=0;
    for(;i<n;)
    {
        if(ar[i]==0)
        {
            int swap=ar[i];
            ar[i]=ar[n-1];
            ar[n-1]=swap;
        }
        n--;
        i++;
    }
    printf("\nUpdated Array : ");
    for(int i=0;i<n2;i++)
    {
        printf("%d ", ar[i]);
    }
}
