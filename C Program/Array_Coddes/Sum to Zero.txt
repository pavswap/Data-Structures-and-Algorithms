#include<stdio.h>
int main()
{
    printf("Enter the size of a Array :");
    int n;int k=0;
    scanf("%d", &n);
    int ar[n];
    for(int i=0;i<n;i++)
    {
        printf("Enter Value to the position %d: ", i);
        scanf("%d", &ar[i]);
    }
    for(int i=0;i<n;i++)
    {
        for(int j=i;j<n;j++)
        {
            if(ar[i]+ar[j]==0)
            {
                printf("%d + %d = 0\n", ar[i], ar[j]);
                k++;
            }
            else
            {
                k=k+0;
            }
        }
    }
    if(k==0)
    {
        printf("There are no such Pairs in the Array");
    }
}
