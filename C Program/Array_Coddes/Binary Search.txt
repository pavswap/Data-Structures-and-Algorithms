#include<stdio.h>
 int main()
 {
    int n;
    printf("Enter the Size of the Array :");
    scanf("%d", &n);
    int ar[n];
    for(int i=0;i<n;i++)
    {
        printf("Enter the Value for index number %d :", i);
        scanf("%d", &ar[i]);
    }
    for(int i=0;i<n-1;i++)
    {
        for(int j=0;j<n-i-1;j++)
        {
            if(ar[j]>ar[j+1])
            {
                int swap=ar[j];
                ar[j]=ar[j+1];
                ar[j+1]=swap;
            }
        }
    }
    printf("\n");
    printf("Sorted Array :");
    for(int i=0;i<n;i++)
    {
        printf("%d ", ar[i]);
    }
    printf("\n");
    //Binary Search
    int sn;
    printf("Enter the Search Value : ");
    scanf("%d", &sn);
    int s=0,e=n-1,mid=0,pos=0;
    while(s<=e)
    {
        mid=(s+e)/2;
        if(sn==ar[mid])
        {
            pos=mid;
            break;
        }
        else if(sn<ar[mid])
        {
            e=mid-1;
        }
        else
        {
            s=mid+1;
        }
    }
    if(pos!=0)
    {
        printf("Position of the Search Value: %d", pos);
    }
    else
    {
        printf("Search Value is not in the Array");
    }
 }
