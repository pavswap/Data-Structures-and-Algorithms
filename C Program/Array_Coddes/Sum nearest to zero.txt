#include<stdio.h>
int main()
{
    printf("Enter the Size of the array \n");
    int n;
    scanf("%d", &n);
    int ar[n];
    for(int i=0;i<n;i++)
    {
        printf("Enter Value to position %d of the Array : ", i);
        scanf("%d", &ar[i]);
    }
    printf("\nThe Given array :");
    for(int i=0;i<n;i++)
    {
       printf("%d ", ar[i]);
    }
    printf("\nThe pair of elements which is minimum :");
    for(int i=0;i<n;i++)
    {
        for(int j=i;j<n;j++)
        {
            if(ar[i]+ar[j]==0 || ar[i]+ar[j]==-1 || ar[i]+ar[j]==1 || ar[i]+ar[j]==-2 || ar[i]+ar[j]==2 || ar[i]+ar[j]==-3 || ar[i]+ar[j]==3)
            {
                printf("%d + %d ~ 0\n", ar[i], ar[j]);
            }
        }
    }
}
