#include<stdio.h>
 int main()
 {
  printf("Enter the Value of n\n");
  int n,k=0,f=0;
  scanf("%d", &n);
  int arr[n];
  printf("Enter the Values in the Array\n");
  for(int i=0;i<n;i++)
  {
    printf("Enter Value for element number %d\n", i);
    scanf("%d", &arr[i]);
  }
  printf("Sorted Array\n");
  for(int i=0;i<n-1;i++)
  {
    for(int j=0;j<n-i;j++)
    {
      if (arr[j] > arr[j + 1])
       {
                k=arr[i];
                arr[j]=arr[i];
                arr[j]=k;
                f=1;
       }
    }
  if(f==1)
    {
    break;
    }
  }
  for(int h=0;h<n;h++)
  {
    printf("%d\n", arr[h]);
  }
 }
