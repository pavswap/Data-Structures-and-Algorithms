#include <stdio.h>

int main() {
    int n;
    
    // Input the size of the array
    printf("Enter the size of array\n");
    scanf("%d", &n);
    
    int ar[n];
    
    // Input array elements
    for (int i = 0; i < n; i++) {
        printf("Enter Value for Position [%d]\n", i);
        scanf("%d", &ar[i]);
    }
    
    // Print the given array
    printf("The given Array is: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", ar[i]);
    }
    printf("\n");
    
    // Find and print next bigger elements
    printf("Next bigger elements are:\n");
    for (int i = 0; i < n; i++) {
        int nextGreater = -1; // Default value if no greater element is found
        
        for (int j = i + 1; j < n; j++) {
            if (ar[j] > ar[i]) {
                nextGreater = ar[j]; // Found the next bigger element
                break;
            }
        }
        
        printf("Next bigger element of %d in the array is: %d\n", ar[i], nextGreater);
    }

    return 0;
}
