#include<stdio.h>
int main()
{
    printf("Enter the size of the array\n");
    int n,s;
    scanf("%d", &n);
    int ar[n];
    for(int i=0;i<n;i++)
    {
        printf("Enter Value for index number %d\n", i);
        scanf("%d", &ar[i]);
    }
    for(int i=0;i<n-1;i++)
    {
        for(int j=0;j<n-i-1;j++)
        {
            if(ar[j]>ar[j+1])
            {
                int swap=ar[j];
                ar[j]=ar[j+1];
                ar[j+1]=swap;
            }
        }
    }
    printf("\n");
    printf("Sorted Array :");
    for(int i=0;i<n;i++)
    {
        printf("%d ", ar[i]);
    }
    printf("\n");
    for(int i=1;i<n;i=i+2)
    {
        for(int j=i-1;j>0;j--)
        {
            int swap = ar[i];
            ar[i]=ar[j];
            ar[j]=swap;
        } 
    }
    printf("\n");
    printf("New Array :");
    for(int i=0;i<n;i++)
    {
        printf("%d ", ar[i]);
    }
}
