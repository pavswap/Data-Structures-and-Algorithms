#include<stdio.h>
int main()
{
    printf("Enter the Value of Array\n");
    int n;
    scanf("%d", &n);
    printf("Enter the Values in the array :-\n");
    int ar[n];
    for(int i=0;i<n;i++)
    {
        scanf("%d", &ar[i]);
    }
    printf("Select the element from where you want the array to be rotated\n");
    int e;
    scanf("%d", &e);
    e=e-1;
    int ar2[n];
    for(int i=e;i<n;i++)
    {
        ar2[i-e]=ar[i];
    }
    for(int i=0;i<e;i++)
    {
        ar2[n-e+i]=ar[i];
    }
    for(int i=0;i<n;i++)
    {
        printf("%d ", ar2[i]);
    }
}
