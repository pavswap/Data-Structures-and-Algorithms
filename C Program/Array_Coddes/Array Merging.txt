#include <stdio.h>
int main()
{
    printf("Enter the number of Elements In the First Array\n");
    int n;
    scanf("%d", &n);
    int a[n];
    for(int i=0;i<n;i++)
    {
        printf("Enter the Value : ");
        scanf("%d", &a[i]);
    }
    printf("Enter the number of Elements In the Second Array\n");
    int n2;
    scanf("%d", &n2);
    int b[n2];
    for(int i=0;i<n2;i++)
    {
        printf("Enter the Value : ");
        scanf("%d", &b[i]);
    }
    int n3;
    n3=n+n2;
    int c[n3];
    for(int i=0;i<n;i++)
    {
        c[i]=a[i];
    }
    for(int i=n;i<n3;i++)
    {
        c[i]=b[i-n];
    }
    for(int i=0;i<n3;i++)
    {
        printf("%d\n", c[i]);
    }
}
