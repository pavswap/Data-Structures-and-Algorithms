#include <math.h>
int main()
{

    int n;
    printf("Enter a Number\n");
    scanf("%d", &n);
    for(int i=n;i>0;i--)
    {
        for(int j=i;j>0;j--)
        {
            printf("%d ", i);
        }
        printf("\n");
    }

}
