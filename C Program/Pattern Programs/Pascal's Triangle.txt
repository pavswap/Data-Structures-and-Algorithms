#include<stdio.h>
int main()
{
    printf("Enter the Size of the Pattern\n");
    int n,coef=1;
    scanf("%d", &n);
    for(int i=1;i<=n;i++)
    {
        coef=1;
        printf("\n");
        for(int k=n-i;k>0;k--)
        {
            printf(" ");
        }
        for(int j=1;j<=i;j++)
        {
            printf("%d ", coef);
            coef=coef*(i-j)/j;
        }
    }
}
