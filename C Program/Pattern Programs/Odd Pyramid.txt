#include<stdio.h>
int main()
{
    printf("Enter the Size of the Pattern\n");
    int n,m=1;
    scanf("%d", &n);
    for(int i=1;i<=n;i++)
    {
        printf("\n");
        for(int j=n-i;j>0;j--)
        {
            printf(" ");
        }
        for(int k=1;k<=2*i-1;k++)
        {
            m++;
            printf("*");
        }
    }
}
