#include<stdio.h>
int main()
{
    printf("Enter the Size of the Pattern\n");
    int n,k=1;
    scanf("%d", &n);
    for(int i=n;i>0;i--)
    {
        printf("\n");
        for(int k=0;k<n-i;k++)
        {
            printf(" ");
        }
        for(int j=i;j>0;j--)
        {
            printf("* ");
            k++;
        }
    }
}
