#include<stdio.h>
int main()
{
    printf("Enter the Size of the Pattern\n");
    int n;
    scanf("%d", &n);
    for(int i=0;i<n;i++)
    {
        printf("\n");
        for(int k=n-i;k>0;k--)
        {
            printf(" ");
        }
        for(int j=0;j<i;j++)
        {
            printf("* ");
        }
    }
    for(int i=n;i>0;i--)
    {
        printf("\n");
        for(int k=0;k<n-i;k++)
        {
            printf(" ");
        }
        for(int j=i;j>0;j--)
        {
            printf("* ");
        }
    }
}
