#include<stdio.h>
int main()
{
    printf("Enter a Number\n");
    int n,num1=0,num2=0,k=0,y=0;
    scanf("%d", &n);
    for(int i=2;i<n;i++)
    {
        for(int j=2;j<n;j++)
        {
            if(i+j==n)
            {
                for(int a=2;a<i;a++)
                {
                    k++;
                }
                if(k==0)
                {
                    num1=i;
                }
                k=0;
                for(int b=2;b<j;b++)
                {
                    k++;
                }
                if(k==0)
                {
                    num2=j;
                }
                k=0;
                printf("%d + %d = %d\n", num1, num2, n);
            }
        }
    }
}
