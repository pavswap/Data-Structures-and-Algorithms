import java.util.*;
class Practice {

    static int i = 1;
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        System.out.print("Enter an Array : ");
        int[] ar = Input.input(sc.nextLine());

        boolean ans = SortChecking(ar);
        System.out.println(ans);
    }
    public static boolean SortChecking(int[] ar) {

        if(i == ar.length) {

            return true;
        }
        if(ar[i] > ar[i - 1]) {
            i++;
            return SortChecking(ar);
        }
        return false;
    }
}
