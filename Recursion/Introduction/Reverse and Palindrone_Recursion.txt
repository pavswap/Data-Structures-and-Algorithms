import java.util.*;
class Practice {

    static int s = 0;
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        System.out.print("Enter a Number : ");
        int n = sc.nextInt();

        System.out.println(palindrone(n));

    }
    public static int recursiveReverse(int n) {

        if(n < 1){
            return s;
        }
        s = s * 10 + (n % 10);
        return recursiveReverse(n / 10);
    }
    public static boolean palindrone(int n) {

        return n == recursiveReverse(n);
    }
}
