import java.util.*;
class Practice {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        System.out.println("Sum of N Numbers");

        System.out.print("Enter a Number : ");
        int n = sc.nextInt();
        int s = 0 ;

        nSum(s, n);
    }
    static void nSum(int s, int n) {

        if(s > n) {
            return;
        }
        ++s;
        System.out.println(s);
        nSum(s, n);
    }
}
